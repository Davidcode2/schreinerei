Fix BUG-002: Token refresh cascade failure.

In `frontend/src/lib/api/client.ts`:

1. Add retry constants at top of file:
   ```typescript
   const MAX_REFRESH_RETRIES = 3
   const REFRESH_BASE_DELAY = 1000 // 1 second
   ```

2. Create refreshWithRetry helper function before the ApiClient class:
   ```typescript
   async function refreshWithRetry(
     refreshToken: string,
     attempt: number = 1
   ): Promise<AuthTokens> {
     try {
       return await refreshAccessToken(refreshToken)
     } catch (error) {
       if (attempt >= MAX_REFRESH_RETRIES) {
         throw error
       }
       // Exponential backoff: 1s, 2s, 4s
       const delay = REFRESH_BASE_DELAY * Math.pow(2, attempt - 1)
       console.warn(`Token refresh failed (attempt ${attempt}), retrying in ${delay}ms...`)
       await new Promise(resolve => setTimeout(resolve, delay))
       return refreshWithRetry(refreshToken, attempt + 1)
     }
   }
   ```

3. Update getAccessToken method to use retry and show warning:
   ```typescript
   private async getAccessToken(): Promise<string | null> {
     const { tokens, setTokens, logout } = useAuthStore.getState()

     if (!tokens) return null

     if (Date.now() >= tokens.expires_at - 60000) {
       try {
         // Show warning toast on first refresh attempt
         if (Date.now() >= tokens.expires_at - 120000) {
           toast.warning('Session wird bald ablaufen')
         }
         const newTokens = await refreshWithRetry(tokens.refresh_token)
         setTokens(newTokens)
         return newTokens.access_token
       } catch (error) {
         console.error('Token refresh failed after retries:', error)
         toast.error('Session abgelaufen. Bitte melden Sie sich erneut an.')
         logout()
         return null
       }
     }

     return tokens.access_token
   }
   ```

4. Add toast import at top:
   ```typescript
   import { toast } from 'sonner'
   ```
