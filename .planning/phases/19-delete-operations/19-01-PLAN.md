From src/modules/fleet/api/routes.rs (lines 434-449):
```rust
pub async fn delete_vehicle(
    State(state): State<AppState>,
    auth: AuthenticatedUser,
    Path(id): Path<String>,
) -> Result<impl IntoResponse, AppError> {
    let service = FleetService::new(FleetRepository::new(state.pool));
    let ctx = TenantContext::from_auth(&auth);
    
    let vehicle_id = Uuid::parse_str(&id)
        .map(VehicleId)
        .map_err(|_| AppError::Validation("Invalid vehicle ID".to_string()))?;
    
    service.delete_vehicle(vehicle_id, &ctx).await?;
    
    Ok((StatusCode::OK, Json(serde_json::json!({ "success": true }))))
}
```

From src/modules/fleet/infrastructure/fleet_repository.rs (lines 188-210):
```rust
pub async fn delete_vehicle(
    &self,
    tenant_id: TenantId,
    id: VehicleId,
) -> Result<(), AppError> {
    let result = sqlx::query(
        r#"
        DELETE FROM vehicles
        WHERE id = $1 AND tenant_id = $2
        "#
    )
    .bind(id.0)
    .bind(tenant_id.0)
    .execute(&self.pool)
    .await
    .map_err(|e| AppError::Database(e.to_string()))?;

    if result.rows_affected() == 0 {
        return Err(AppError::NotFound("Vehicle not found".to_string()));
    }

    Ok(())
}
```
