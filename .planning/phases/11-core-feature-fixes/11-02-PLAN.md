pub async fn create_reservation(
    &self,
    create: CreateReservation,
    ctx: &TenantContext,
) -> Result<Reservation, AppError> {
    create.validate()?;

    // Resolve local user ID
    let user_repo = UserRepository::new(self.pool.clone());
    let local_user_id = user_repo
        .find_or_create_by_keycloak_id(
            &ctx.user_id.to_string(),
            ctx.tenant_id,
            &ctx.email,
            if ctx.is_admin() { Role::Admin } else { Role::Employee },
        )
        .await?
        .id;

    let reservation = self.fleet_repo
        .create_reservation(&create, local_user_id, ctx.tenant_id)
        .await?;
