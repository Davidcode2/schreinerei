    ```rust
    use crate::common::error::AppError;
    use crate::common::types::{TenantId, UserId, SiteId};
    use crate::modules::iam::domain::user_preferences::{UserPreferences, UserPreferenceRecord, UpdatePreferences};
    use crate::modules::iam::infrastructure::user_preferences_repository::UserPreferencesRepository;
    use crate::modules::sites::infrastructure::site_repository::SiteRepository;
    use sqlx::PgPool;

    pub struct UserPreferencesService {
        preferences_repo: UserPreferencesRepository,
        site_repo: SiteRepository,
    }

    impl UserPreferencesService {
        pub fn new(pool: PgPool) -> Self {
            Self {
                preferences_repo: UserPreferencesRepository::new(pool.clone()),
                site_repo: SiteRepository::new(pool),
            }
        }

        /// Get user's preferences
        pub async fn get_preferences(
            &self,
            user_id: UserId,
            tenant_id: TenantId,
        ) -> Result<UserPreferenceRecord, AppError>

        /// Set active site with validation
        /// Returns error if site doesn't exist, doesn't belong to tenant, or is archived/deleted
        pub async fn set_active_site(
            &self,
            user_id: UserId,
            tenant_id: TenantId,
            site_id: SiteId,
        ) -> Result<UserPreferenceRecord, AppError>

        /// Clear active site
        pub async fn clear_active_site(
            &self,
            user_id: UserId,
            tenant_id: TenantId,
        ) -> Result<UserPreferenceRecord, AppError>

        /// Validate and auto-clear if active site is invalid
        /// Called when retrieving preferences to ensure validity
        pub async fn get_validated_preferences(
            &self,
            user_id: UserId,
            tenant_id: TenantId,
        ) -> Result<UserPreferenceRecord, AppError>
    }
    ```

    Implementation details for `set_active_site`:
    1. Query site_repo.find_site_by_id(tenant_id, site_id)
    2. If None, return AppError::NotFound("Site not found")
    3. If site.status is Archived, return AppError::Validation("Cannot set archived site as active")
    4. If site.deleted_at is not null, return AppError::NotFound("Site not found")
    5. Update preferences with the site_id

    Implementation details for `get_validated_preferences`:
    1. Get preferences
    2. If active_site_id is Some, validate the site still exists and is not archived/deleted
    3. If invalid, auto-clear by calling clear_active_site and return updated preferences
    4. Return preferences

    Then update `src/modules/iam/application.rs` to add:
